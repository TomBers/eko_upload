00139 {
00140  unsigned char idx = 0;
00141  unsigned char thresh = 8;
00142  unsigned char temp;
00143  
00144  // wait until data available bit is set
00145  while ( !U1STAbits.URXDA  ) {}
00146  
00147  // Use TMR2 for timeout of around 10ms
00148  TMR2 = 0;
00149  T2CON = 0x8030; // 1:256 prescale, enable tmr2
00150  
00151   while ( (idx < thresh) && (TMR2 < 20) && (idx < RX_BUF_MAX) )
00152  {
00153   // empty buffer while data available
00154   while(U1STAbits.URXDA)
00155   {
00156    rxbuf[idx++] = (unsigned char)(0x00FF & U1RXREG);
00157    TMR2 = 0;
00158   }
00159   // check for function code 16
00160   if ((idx == 7) && (rxbuf[2] == 0x10))
00161   {
00162    // set pdu length for extra bytes
00163    thresh = 9 + rxbuf[6];
00164   }
00165   
00166  }
00167  
00168  // disable timer
00169  T2CON = 0x0000;
00170 
00171  // flush buffer of extra data
00172  while( U1STAbits.URXDA )
00173  {
00174   temp = (unsigned char)(0x00FF & U1RXREG); // dump extra characters
00175  }
00176  U1STA = 0x0400;
