00130 // Description: Calculate the approximate illuminance (lux) given the raw
00131 // channel values of the TSL2560. The equation if implemented
00132 // as a piece-wise linear approximation.
00133 //
00134 // Arguments: unsigned int iGain - gain, where 0:1X, 1:16X
00135 // unsigned int tInt - integration time, where 0:13.7mS, 1:100mS, 2:402mS,
00136 // 3:Manual
00137 // unsigned int ch0 - raw channel value from channel 0 of TSL2560
00138 // unsigned int ch1 - raw channel value from channel 1 of TSL2560
00139 // unsigned int iType - package type (T or CS)
00140 //
00141 // Return: unsigned int - the approximate illuminance (lux)
00142 //
00144 unsigned int CalculateLux(unsigned int iGain, unsigned int tInt, unsigned int ch0, 
00145 unsigned int ch1, int iType)
00146 {
00147 //------------------------------------------------------------------------
00148 // first, scale the channel values depending on the gain and integration time
00149 // 16X, 402mS is nominal.
00150 // scale if integration time is NOT 402 msec
00151 unsigned long chScale;
00152 unsigned long channel1;
00153 unsigned long channel0;
00154 switch (tInt)
00155 {
00156 case 0: // 13.7 msec
00157 chScale = CHSCALE_TINT0;
00158 break;
00159 case 1: // 101 msec
00160 chScale = CHSCALE_TINT1;
00161 break;
00162 default: // assume no scaling
00163 chScale = (1 << CH_SCALE);
00164 
00165 break;
00166 }
00167 // scale if gain is NOT 16X
00168 if (!iGain) chScale = chScale << 4; // scale 1X to 16X
00169 // scale the channel values
00170 channel0 = (ch0 * chScale) >> CH_SCALE;
00171 channel1 = (ch1 * chScale) >> CH_SCALE;
00172 //------------------------------------------------------------------------
00173 // find the ratio of the channel values (Channel1/Channel0)
00174 // protect against divide by zero
00175 unsigned long ratio1 = 0;
00176 if (channel0 != 0) ratio1 = (channel1 << (RATIO_SCALE+1)) / channel0;
00177 // round the ratio value
00178 unsigned long ratio = (ratio1 + 1) >> 1;
00179 // is ratio <= eachBreak ?
00180 unsigned int b, m;
00181 switch (iType)
00182 {
00183 case 0: // T package
00184 if ((ratio >= 0) && (ratio <= K1T))
00185 {b=B1T; m=M1T;}
00186 else if (ratio <= K2T)
00187 {b=B2T; m=M2T;}
00188 else if (ratio <= K3T)
00189 {b=B3T; m=M3T;}
00190 else if (ratio <= K4T)
00191 {b=B4T; m=M4T;}
00192 else if (ratio <= K5T)
00193 {b=B5T; m=M5T;}
00194 else if (ratio <= K6T)
00195 {b=B6T; m=M6T;}
00196 else if (ratio <= K7T)
00197 {b=B7T; m=M7T;}
00198 else if (ratio > K8T)
00199 {b=B8T; m=M8T;}
00200 break;
00201 case 1:// CS package
00202 if ((ratio >= 0) && (ratio <= K1C))
00203 {b=B1C; m=M1C;}
00204 else if (ratio <= K2C)
00205 {b=B2C; m=M2C;}
00206 else if (ratio <= K3C)
00207 {b=B3C; m=M3C;}
00208 else if (ratio <= K4C)
00209 {b=B4C; m=M4C;}
00210 else if (ratio <= K5C)
00211 {b=B5C; m=M5C;}
00212 else if (ratio <= K6C)
00213 {b=B6C; m=M6C;}
00214 else if (ratio <= K7C)
00215 {b=B7C; m=M7C;}
00216 
00217 else if (ratio > K8C)
00218 {b=B8C; m=M8C;}
00219 break;
00220 }
00221 unsigned long temp;
00222 temp = ((channel0 * b) - (channel1 * m));
00223 // do not allow negative lux value
00224 if (temp < 0) temp = 0;

00226 temp += (1 << (LUX_SCALE-1));
