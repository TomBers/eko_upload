00542 /*********************************************************************
00543 * Function:        EEAckPolling()
00544 *
00545 * Input:  Control byte.

00547 * Output:  error state.

00549 * Overview:  polls the bus for an Acknowledge from device
00550 *
00551 * Note:   None
00552 ********************************************************************/
00553 unsigned int EEAckPolling(unsigned char control)
00554 {
00555  IdleI2C();    //wait for bus Idle
00556  StartI2C();    //Generate Start condition
00557  
00558  if(I2C1STATbits.BCL)
00559  {
00560   return(-1);   //Bus collision, return
00561  }
00562 
00563  else
00564  {
00565   if(WriteI2C(control))
00566   {
00567    return(-3);  //error return
00568   }
00569 
00570   IdleI2C();   //wait for bus idle
00571   if(I2C1STATbits.BCL)
00572   {
00573    return(-1);  //error return
00574   }
00575 
00576   while(!ACKStatus())
00577   {
00578    RestartI2C(); //generate restart
00579    if(I2C1STATbits.BCL)
00580    {
00581     return(-1); //error return
00582    }
00583 
00584    if(WriteI2C(control))
00585    {
00586     return(-3);
00587    }
00588 
00589    IdleI2C();
00590   }
00591  }
00592  StopI2C();    //send stop condition
00593  if(I2C1STATbits.BCL)
00594  {
00595   return(-1);
00596  }
00597  return(0);
00598 }
